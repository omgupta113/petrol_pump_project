    `Petrol Pump ID` VARCHAR(255) UNIQUE,
    `Name` VARCHAR(255),
    `Location` VARCHAR(255)
);

CREATE TABLE `Petrol Pump Detail` (
    `Petrol Pump ID` VARCHAR(255),
    `Vehicle ID` VARCHAR(255),
    `Entering Time` VARCHAR(255),
    `Exit Time` VARCHAR(255),
    `Filling Time` VARCHAR(255),
    `Date` DATE,
    FOREIGN KEY (`Petrol Pump ID`) REFERENCES `Petrol Pump`(`Petrol Pump ID`)
);
